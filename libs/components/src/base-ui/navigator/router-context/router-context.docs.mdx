import { RouterProvider } from './router-context';
// import { Link } from '@watheia/base-ui.navigator.link';

Defines the structure of our routing system,  
and provides the underlying implementation of links in the project.

As simple as:

```tsx
import { reactRouterAdapter } from '@teambit/ui-foundation.ui.navigation.react-router.routing-adapter';
import { RoutingProvider } from '@watheia/base-ui.navigator.router-context';

function App() {
  return <RoutingProvider value={reactRouterAdapter}>{children}</RoutingProvider>;
}
```
