import { MemoryRouter } from 'react-router-dom';
import { Link, NavLink } from '@teambit/base-react.navigation.link';

A React Router adapter for our routing system. Any `<Link>` rendered under this will have the React Router implementation!

Simply wrap you app with `<ReactRouterRoutingProvider>`, like so:

```tsx
function AppContext({ children }) {
  return <ReactRouterRoutingProvider>{children}</ReactRouterRoutingProvider>;
}
```

Try it out!

```tsx
() => {
  const Example = () => {
    const location = useLocation();

    return (
      <div>
        <div>current location is: {JSON.stringify(location, undefined, 1)}</div>
        <Link href="/some/link">/some/link</Link>
        <br />
        <NavLink href="/some/link" activeStyle={{ fontWeigth: 'bold' }}>
          /another/link
        </NavLink>
      </div>
    );
  };

  return (
    <MemoryRouter> {/* <- required by React Router */}
      <ReactRouterRoutingProvider>
        <Example />
      </ReactRouterRoutingProvider>
    </MemoryRouter>
  );
};
```
